Шаблоны и data binding

index.html


<!DOCTYPE html>
<html ng-app>
  <head>
    <script src="angular.min.js"></script>
  </head>
  <body ng-init="name = 'World'">
    <h1>Hello {{ name }}!</h1>
    <input ng-model="name" autofocus />
  </body>
</html>

Scope

  • клей между контроллером и шаблоном
  • объект, ссылающийся на модели приложения
  • контекст выполнения выражений в шаблоне
  • один $rootScope и дерево дочерних scope

Контроллер

  • контроллер — код описывающий поведение шаблона
  • контроллер создает новый изолированный scope

index.html


<!DOCTYPE html>
<html ng-app="dcApp">
  <head>
    <script src="angular.min.js"></script>
    <script src="script.js"></script>
  </head>
  <body>
    <div ng-controller="FirstCtrl">
      <h1>Hello {{ name }}!</h1>
      <input ng-model="name" autofocus />
    </div>
    <div ng-controller="SecondCtrl">
      <h1>Hello {{ name }}!</h1>
      <input ng-model="name" />
    </div>
  </body>
</html>

script.js


var dcApp = angular.module("dcApp", []);

dcApp.controller("FirstCtrl", function ($scope) {
  $scope.name = "Ivan";
});

dcApp.controller("SecondCtrl", function ($scope) {
  $scope.name = "Petr";
});

Директивы

  • описывают поведение или изменение DOM элемента
  • позволяют расширять HTML в декларативном стиле
  • вызываются через аттрибут, название элемента, CSS класс или комментарий

index.html


<!DOCTYPE html>
<html ng-app="dcApp">
  <head>
    <script src="angular.min.js"></script>
    <script src="script.js"></script>
  </head>
  <body>
    <hello-name name="Ivan" autofocus></hello-name>
    <hello-name name="Petr"></hello-name>
  </body>
</html>

script.js


var dcApp = angular.module("dcApp", []);

dcApp.directive("helloName", function () {
  return {
    scope: {
      name: "@"
    },
    restrict: "E",
    templateUrl: "hello-name.html",
    replace: true,
    link: function (scope, element, attrs) {
      if (attrs.hasOwnProperty("autofocus")) {
        element.find("input")[0].focus()
      }
    }
  };
});
